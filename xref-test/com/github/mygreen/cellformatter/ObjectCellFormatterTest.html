36      @Test
37      public void test_format_numeric_double() {
38          
39          ObjectCellFormatter cellFormatter = new ObjectCellFormatter();
40          
41          ObjectCell<?> cell = new NumberCell<Double>(-1234.5d, "#,##0.0_);[Red]\\(#,##0.0\\)");
42          
43          CellFormatResult result = cellFormatter.format(cell);
44          
45          assertThat(result.getCellType(), is(FormatCellType.Number));
46          assertThat(result.getText(), is("(1,234.5)"));
47          assertThat(result.getTextColor(), is(MSColor.RED));
48          assertThat(result.getSectionPattern(), is("[Red]\\(#,##0.0\\)"));
49          assertThat(result.getValueAsDoulbe(), is(-1234.5));
50      }

86      @Test
87      public void test_format_num() {
88          
89          ObjectCellFormatter cellFormatter = new ObjectCellFormatter();
90          
91          ObjectCell<?> cell = new NumberCell<Double>(0.000012345d, "0.000E+00");
92          
93          CellFormatResult result = cellFormatter.format(cell);
94          
95          assertThat(result.getCellType(), is(FormatCellType.Number));
96          
97          if(IS_JAVA_1_8) {
98              assertThat(result.getText(), is("1.234E-05"));
99          } else {
100             assertThat(result.getText(), is("1.235E-05"));
101             
102         }

106     @Test
107     public void test_format_date() {
108         
109         ObjectCellFormatter cellFormatter = new ObjectCellFormatter();
110         
111         ObjectCell<?> cell = new DateCell(Timestamp.valueOf("2012-02-29 13:01:40"), "[$-411]ggge\"年\"m\"月\"d\"日\"\\ h:mm\\ AM/PM;@");
112         
113         CellFormatResult result = cellFormatter.format(cell);
114         
115         assertThat(result.getCellType(), is(FormatCellType.Date));
116         assertThat(result.getText(), is("平成24年2月29日 1:01 午後"));
117         assertThat(result.getTextColor(), is(nullValue()));
118         assertThat(result.getSectionPattern(), is("[$-411]ggge\"年\"m\"月\"d\"日\"\\ h:mm\\ AM/PM"));
119         assertThat(result.getValueAsDate(), is(new Date(Timestamp.valueOf("2012-02-29 13:01:40.000").getTime())));
120         
121     }
