29      @Test
30      public void testTime0800() throws Exception{
31  
32          final Sheet sheet = getTestSheet();
33          SimpleDateFormat dateFormatter = new SimpleDateFormat("yyyy-MM-dd HH:mm:ss");
34  
35          Cell cell = sheet.getRow(2).getCell(2);
36          String address = formatCellAddress(cell);
37  
38          {
39              Date date = cell.getDateCellValue();
40              String formattedDate = dateFormatter.format(date);
41  
42              assertThat(formattedDate, is("1899-12-31 08:00:00"));
43              System.out.printf("[%s][読み込み時]date=%s\n", address, formattedDate);
44          }
45  
46          {
47              // 日時を数値に変換
48              Date date = cell.getDateCellValue();
49              Date date2 = stripTimeZone(date);
50  
51              String formattedDate = dateFormatter.format(date);
52              String formattedDate2 = dateFormatter.format(date2);
53  
54              assertThat(formattedDate2, is("1899-12-31 17:00:00"));
55  
56              System.out.printf("[%s]date=%s, GMT=%s\n", address, formattedDate, formattedDate2);
57  
58              double num = ExcelDateUtils.convertExcelNumber(date2, false);
59              cell.setCellValue(num);
60              Date date3 = cell.getDateCellValue();
61              String formattedDate3 = dateFormatter.format(date3);
62              assertThat(formattedDate3, is("1899-12-31 08:00:00"));
63  
64              System.out.printf("[%s][変換時]num=%f,date=%s\n", address, num, formattedDate3);
65          }
66  
67  
68      }
69  
70      private Sheet getTestSheet() throws Exception {
71  
72          try (InputStream in = Files.newInputStream(Paths.get("src/test/data", "cell_time.xlsx"))) {
73              Workbook book = WorkbookFactory.create(in);
74  
75              Sheet sheet = book.getSheet("時間テスト");
76              return sheet;
77  
78          }
79  
80      }

82      /**
83       * セルのアドレス'A1'を取得する。
84       * @param cell セル情報
85       * @return IllegalArgumentException cell == null.
86       */
87      public static String formatCellAddress(final Cell cell) {
88          ArgUtils.notNull(cell, "cell");
89          return CellReference.convertNumToColString(cell.getColumnIndex()) + String.valueOf(cell.getRowIndex()+1);
90      }
91  
92      private Date stripTimeZone(Date date) {
93  
94          return new Date(date.getTime() + TimeZone.getDefault().getRawOffset());
95  
96      }
