19      public void testGetIndex() {
20          assertThat(MSColor.BLACK.getIndex(), is((short)1));
21          assertThat(MSColor.WHITE.getIndex(), is((short)2));
22          assertThat(MSColor.RED.getIndex(), is((short)3));
23          assertThat(MSColor.GREEN.getIndex(), is((short)4));
24          assertThat(MSColor.BLUE.getIndex(), is((short)5));
25          assertThat(MSColor.YELLOW.getIndex(), is((short)6));
26          assertThat(MSColor.MAGENTA.getIndex(), is((short)7));
27          assertThat(MSColor.CYAN.getIndex(), is((short)8));
28      }

30      @Test
31      public void testGetName() {
32          assertThat(MSColor.BLACK.getName(), is("Black"));
33          assertThat(MSColor.WHITE.getName(), is("White"));
34          assertThat(MSColor.RED.getName(), is("Red"));
35          assertThat(MSColor.GREEN.getName(), is("Green"));
36          assertThat(MSColor.BLUE.getName(), is("Blue"));
37          assertThat(MSColor.YELLOW.getName(), is("Yellow"));
38          assertThat(MSColor.MAGENTA.getName(), is("Magenta"));
39          assertThat(MSColor.CYAN.getName(), is("Cyan"));
40      }

42      @Test
43      public void testGetLocaleNameLocale() {
44          assertThat(MSColor.BLACK.getLocaleName(Locale.JAPANESE), is("黒"));
45          assertThat(MSColor.WHITE.getLocaleName(Locale.JAPANESE), is("白"));
46          assertThat(MSColor.RED.getLocaleName(Locale.JAPANESE), is("赤"));
47          assertThat(MSColor.GREEN.getLocaleName(Locale.JAPANESE), is("緑"));
48          assertThat(MSColor.BLUE.getLocaleName(Locale.JAPANESE), is("青"));
49          assertThat(MSColor.YELLOW.getLocaleName(Locale.JAPANESE), is("黄"));
50          assertThat(MSColor.MAGENTA.getLocaleName(Locale.JAPANESE), is("紫"));
51          assertThat(MSColor.CYAN.getLocaleName(Locale.JAPANESE), is("水"));
52          
53          // 定義されていないロケール
54          assertThat(MSColor.BLACK.getLocaleName(Locale.CHINESE), is("Black"));
55      }

57      @Test
58      public void testGetHtmlColor() {
59          assertThat(MSColor.BLACK.getHtmlColor(), is("#000000"));
60          assertThat(MSColor.WHITE.getHtmlColor(), is("#FFFFFF"));
61          assertThat(MSColor.RED.getHtmlColor(), is("#FF0000"));
62          assertThat(MSColor.GREEN.getHtmlColor(), is("#00FF00"));
63          assertThat(MSColor.BLUE.getHtmlColor(), is("#0000FF"));
64          assertThat(MSColor.YELLOW.getHtmlColor(), is("#FFFF00"));
65          assertThat(MSColor.MAGENTA.getHtmlColor(), is("#FF00FF"));
66          assertThat(MSColor.CYAN.getHtmlColor(), is("#00FFFF"));
67      }

69      @Test
70      public void testGetRgbColor() {
71          assertThat(MSColor.BLACK.getRgbColor(), is(new short[]{0,0,0}));
72          assertThat(MSColor.WHITE.getRgbColor(), is(new short[]{255,255,255}));
73          assertThat(MSColor.RED.getRgbColor(), is(new short[]{255, 0, 0}));
74          assertThat(MSColor.GREEN.getRgbColor(), is(new short[]{0, 255, 0}));
75          assertThat(MSColor.BLUE.getRgbColor(), is(new short[]{0, 0, 255}));
76          assertThat(MSColor.YELLOW.getRgbColor(), is(new short[]{255, 255, 0}));
77          assertThat(MSColor.MAGENTA.getRgbColor(), is(new short[]{255, 0, 255}));
78          assertThat(MSColor.CYAN.getRgbColor(), is(new short[]{0, 255, 255}));
79      }
