1   package com.github.mygreen.cellformatter;

3   import java.util.Locale;
4   import java.util.Map;
5   import java.util.concurrent.ConcurrentHashMap;
6   
7   import com.github.mygreen.cellformatter.lang.ArgUtils;
8   
9   
10  /**
11   * ロケールによって、フォーマッタを切り替えて処理する。
12   * @author T.TSUCHIE
13   *
14   */
15  public class LocaleSwitchFormatter extends CellFormatter {
16      
17      private final CellFormatter defaultFormatter;
18      
19      private final Map<Locale, CellFormatter> formatterMap = new ConcurrentHashMap<>();
20      
21      public LocaleSwitchFormatter(final CellFormatter defaultFormatter) {
22          ArgUtils.notNull(defaultFormatter, "defaultFormatter");
23          this.defaultFormatter = defaultFormatter;
24      }
25      
26      @Override
27      public CellFormatResult format(final CommonCell cell, final Locale locale) {
28          
29          if(locale == null) {
30              return defaultFormatter.format(cell, locale);
31              
32          } else if(formatterMap.containsKey(locale)) {
33              return formatterMap.get(locale).format(cell, locale);
34          }
35          
36          return defaultFormatter.format(cell, locale);
37      }
38      
39      /**
40       * ローケールとフォーマッタを登録する。
41       * @param cellFormatter
42       * @param locales
43       * @return
44       * @throws IllegalArgumentException cellFormatter is null.
45       * @throws IllegalArgumentException locales size is zero.
46       */
47      public LocaleSwitchFormatter register(final CellFormatter cellFormatter, final Locale... locales) {
48          ArgUtils.notNull(cellFormatter, "cellFormatter");
49          ArgUtils.notEmpty(locales, "locales");
50          
51          for(Locale locale : locales) {
52              formatterMap.put(locale, cellFormatter);
53          }
54          
55          return this;
56      }
57      
58  }
