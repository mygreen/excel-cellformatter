9   /**
10   *  フォーマットの書式'*'の記号を表現する項。
11   * ・'*'の次に続く文字をセルの幅によって埋める。
12   * ・通常は何もしない。
13   * 
14   * @author T.TSUCHIE
15   *
16   */
17  public class AsteriskTerm<T> implements Term<T> {
18      
19      private static final String EMPTY = "";
20      
21      private final Token.Asterisk token;
22      
23      public AsteriskTerm(final Token.Asterisk token) {
24          this.token = token;
25      }
26      
27      @Override
28      public String format(final T number, final MSLocale formatLocale, final Locale runtimeLocale) {
29          return EMPTY;
30      }
31      
32      public Token.Asterisk getToken() {
33          return token;
34      }
35  }
