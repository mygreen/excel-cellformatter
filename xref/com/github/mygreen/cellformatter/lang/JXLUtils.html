25       * @param workbook 判定対象のワークブック。
26       * @return true:1904年始まり
27       * @throws IllegalArgumentException {@literal workbook == null.}
28       */
29      public static boolean isDateStart1904(final Workbook workbook) {
30          ArgUtils.notNull(workbook, "workbook");
31          
32          if(workbook instanceof WorkbookParser) {
33              try {
34                  Field field = WorkbookParser.class.getDeclaredField("nineteenFour");
35                  field.setAccessible(true);
36                  
37                  boolean value = field.getBoolean(workbook);
38                  return value;
39                  
40              } catch (NoSuchFieldException | SecurityException e) {
41                  logger.warn("fail access field WrokbookParser#nineteenFour", e);
42                  return false;
43                  
44              } catch (IllegalArgumentException | IllegalAccessException e) {
45                  logger.warn("fail invoke field WrokbookParser#nineteenFour", e);
46                  return false;
47              }
48              
49          }
50          
51          return false;
52      }
