4   import java.text.ParseException;
5   import java.text.SimpleDateFormat;
6   import java.util.Date;

10  /**
11   * Excelの日時表現を処理するためのユーティリティクラス。
12   * <ul>
13   *  <li>Excel上は、1900年は閏年扱いで、1900年2月29日が存在する。</li>
14   *  <li>Javaの形式に変換したときは、1900年3月1日扱いとする。</li>
15   * </ul>
16   *
17   * @since 0.6
18   * @author T.TSUCHIE
19   *
20   */
21  public class ExcelDateUtils {
22      
23      /**
24       * 24時間の秒数。
25       */
26      public static final int SECONDS_IN_DAYS = (int) TimeUnit.HOURS.toSeconds(24);
27      
28      /**
29       * {@literal 1900-01-01 00:00:00.000}の時間（単位はミリ秒）。
30       * <p>Excelは設定により、1900年始まりか1904年始まりか指定できるため、その基準値として利用する。
31       */
32      public static final long MILLISECONDS_19000101 = parseDate("1900-01-01 00:00:00.000").getTime();
33      
34      /**
35       * {@literal 1904-01-01 00:00:00.000}の時間（単位はミリ秒）。
36       * <p>Excelは設定により、1900年始まりか1904年始まりか指定できるため、その基準値として利用する。
37       */
38      public static final long MILLISECONDS_19040101 = parseDate("1904-01-01 00:00:00.000").getTime();
39      
40      /**
41       * 1900-03-01の時間（単位はミリ秒）。
42       * <p>Excelは1900年始まりの場合、閏日でない1900年2月29日（=3月1日）が存在するため、その基準値として利用する。
43       */
44      public static final long MILLISECONDS_19000301 = ExcelDateUtils.parseDate("1900-03-01 00:00:00.000").getTime();
45      
46      /**
47       * Javaの基準日(=0)である1970年1月1日に対するExcelの基準日1900年1月0日の日のオフセット（単位は日）。
48       * <p>ただし、Excelは1900年は1月0日から始まり、実質1899年12月31日となる。
49       */
50      public static final int OFFSET_DAYS_1900; 
51      
52      /**
53       * Javaの基準日(=0)である1970年1月1日に対するExcelの基準日1904年1月1日の日のオフセット（単位は日）。
54       */
55      public static final int OFFSET_DAYS_1904;
56      
57      static {
58          long offsetDay1900 = parseDate("1900-01-01 00:00:00.000").getTime() / ((long)SECONDS_IN_DAYS * 1000);
59          OFFSET_DAYS_1900 = (int) offsetDay1900 -1;
60          
61          long offsetDay1904 = parseDate("1904-01-01 00:00:00.000").getTime() / ((long)SECONDS_IN_DAYS * 1000);
62          OFFSET_DAYS_1904 = (int) offsetDay1904;
63      }
64      
65      /**
66       * 1900年開始の場合、3月1日の経過日数。
67       * <p>1900年は閏年ではないが、Excelの場合は閏年扱いのため、
68       *    1900年1月～2月の期間（60日23時59分59秒=60.9999）を表現するための定数として利用する。
69       */
70      public static final int NON_LEAP_DAY = 61;
71      
72      /**
73       * Excel表現上の数値をJavaの{@link Date}型(UTC形式)に変換する。
74       * <p>1900年始まりの場合は以下の注意が必要。</p>
75       * <ul>
76       *   <li>値{@literal 0.0}は、Excel上は1900年1月0日であるが、Date型へ変換した場合は1899年12月31日となります。</li>
77       *   <li>値{@literal 60.0}は、Excel上は1900年2月29日だが、グレゴリオ歴上は閏日ではあにため、1900年3月1日となります。</li>
78       * </ul>
79       * 
80       * @param numValue 変換対象のExcel表現上の数値。
81       * @param startDate1904 基準日が1904年始まりかどうか。
82       * @return Java表現上に変換した日時。
83       *         ただし、この値はタイムゾーンは考慮されていない（=GMT-00:00）ため、
84       *         変換後は独自に処理を行う必要があります。
85       */
86      public static Date convertJavaDate(final double numValue, final boolean startDate1904) {
87          
88          double utcDay;
89          if(startDate1904) {
90              // 1904年始まりの場合
91              
92              // UTC(1970年基準に戻す)
93              utcDay = numValue + OFFSET_DAYS_1904;
94              
95          } else {
96              // 1900年始まりの場合
97              
98              // UTC(1970年基準に戻す)
99              utcDay = numValue + OFFSET_DAYS_1900;
100             
101             /*
102              * 1900年3月1日（Excel上は1900年2月29日）以降の場合の補正。
103              * ・Excel上は1900年は、閏年扱いで1900年2月29日の閏年が存在する。
104              * ・しかし、グレゴリオ歴上は1900年は閏年ではないため、UTCの表現上から見ると1日多い。
105              * ・1900年2月29日は、1900年1月0日から数えて61日目なので、61日以降は、1日前に戻す。
106              */
107             if(numValue >= NON_LEAP_DAY) {
108                 utcDay -= 1;
109             }
110             
111         }
112         
113         /*
114          * Javaのミリ秒に直す。
115          * ・Excelの日付の形式の場合小数部が時間を示すため、24時間分のミリ秒を考慮する。
116          */
117         long utcTime = Math.round(utcDay * SECONDS_IN_DAYS) * 1000;
118         return new Date(utcTime);
119         
120     }

122     /**
123      * Javaの{@link Date}型をExcelの内部表現の数値に変換する。
124      * <p>小数の桁数に関する注意事項。</p>
125      * <ul>
126      *    <li>このメソッドは少数は第16位まで保証し、小数第17位は四捨五入して計算します。</li>
127      *    <li>Excelの1秒は、UTC上では1/(60x60x24x1000)=0.0000000115741=1.15741e-008であるので、小数13位までの精度が必要。</li>
128      *    <li>Excelは秒までだが、Javaはミリ秒まで存在するので、さらに3桁多い、16桁までの精度が必要になる。</li>
129      * </ul>
130      * 
131      * <p>1900年始まりの場合は以下の注意が必要。</p>
132      * <ul>
133      *   <li>UTC上は1900年2月29日は存在しないため、{@literal 60.0}への変換はできません。</li>
134      * </ul>
135      * 
136      * @param value 変換対象のJava表現上の日時。タイムゾーンを排除した（GMT-00:00）日時。
137      * @param startDate1904 基準日が1904年始まりかどうか。
138      * @return Excel表現上に変換した数値。
139      * @throws IllegalArgumentException {@literal value == nulll.}
140      */
141     public static double convertExcelNumber(final Date value, final boolean startDate1904) {
142         
143         ArgUtils.notNull(value, "value");
144         
145         /*
146          * Excelの時間の表現に直す。
147          * ・Excelの日付の形式の場合小数部が時間を示すため、24時間分のミリ秒を考慮する。
148          */
149         long utcDay = value.getTime();
150         BigDecimal numValue = new BigDecimal(utcDay);
151         numValue = numValue.divide(new BigDecimal(SECONDS_IN_DAYS * 1000), 17, BigDecimal.ROUND_HALF_UP);
152         
153         if(startDate1904) {
154             // 1904年始まりの場合
155             numValue = numValue.subtract(new BigDecimal(OFFSET_DAYS_1904));
156             
157         } else {
158             // 1900年始まりの場合
159             numValue = numValue.subtract(new BigDecimal(OFFSET_DAYS_1900));
160             
161             if(numValue.compareTo(new BigDecimal(NON_LEAP_DAY - 1)) >= 0) {
162                 numValue = numValue.add(new BigDecimal(1));
163             }
164             
165         }
166         
167         return numValue.doubleValue();
168         
169     }

171     /**
172      * 日時形式を{@literal yyyy-MM-dd HH:mm:ss.SSS}の書式でフォーマットする。
173      * <p>ただし、タイムゾーンは、標準時間の{@literal GMT-00:00}で処理する。
174      * @param date フォーマット対象の日時。
175      * @return フォーマットした文字列。
176      * @throws IllegalArgumentException date is null.
177      */
178     public static String formatDate(final Date date) {
179         
180         ArgUtils.notNull(date, "date");
181         
182         SimpleDateFormat format = new SimpleDateFormat("yyyy-MM-dd HH:mm:ss.SSS");
183         format.setTimeZone(TimeZone.getTimeZone("GMT-00:00"));
184         return format.format(date);
185         
186     }
187     
188     /**
189      * 文字列を日時形式を{@literal yyyy-MM-dd HH:mm:ss.SSS}のパースする。
190      * <p>ただし、タイムゾーンは、標準時間の{@literal GMT-00:00}で処理する。
191      * @param str パース対象の文字列
192      * @return パースした日付。
193      * @throws IllegalArgumentException str is empty.
194      * @throws IllegalStateException fail parsing.
195      */
196     public static Date parseDate(final String str) {
197         ArgUtils.notEmpty(str, str);
198         
199         try {
200             SimpleDateFormat format = new SimpleDateFormat("yyyy-MM-dd HH:mm:ss.SSS");
201             format.setTimeZone(TimeZone.getTimeZone("GMT-00:00"));
202             return format.parse(str);
203         } catch (ParseException e) {
204             throw new IllegalStateException(String.format("fail parse to Data from '%s',", str), e);
205         }
206         
207     }

209     /**
210      * Excelの日付となる（=数値では0の値）の時の、時間の取得。（単位はミリ秒）。
211      * <p>1900年始まりの場合、Excelでは1月0日から始まるため、{@literal 1899-12-31 00:00:00.000}の値を返す。
212      * <p>1904年始まりは、{@literal 1904-01-01 0:00:00}の値を返す。
213      * <p>引数により、1904年始まりの場合の値か選択できる。
214      * @param isStartDate1904 1904年始まりかどうか。
215      * @return Excelの基準日に対するUTC表現上のミリ秒。
216      */
217     public static long getExcelZeroDateTime(boolean isStartDate1904) {
218         if(isStartDate1904) {
219             return MILLISECONDS_19040101;
220         } else {
221             return MILLISECONDS_19000101 - TimeUnit.DAYS.toMillis(1);
222         }
223     }
