10  /**
11   * セルのフォーマッタを解決するクラス。
12   * <p>解析したフォーマットをキャッシュし、性能を向上する。
13   * @author T.TSUCHIE
14   *
15   */
16  public class FormatterResolver {
17      
18      private static final MessageResolver messageResolver = new MessageResolver("com.github.mygreen.cellformatter.format");
19      
20      /**
21       * カスタム書式のインスタンスを作成する。
22       */
23      private CustomFormatterFactory customFormatterFactory = new CustomFormatterFactory();
24      
25      /**
26       * 書式のインデックスとフォーマッターのマップ
27       */
28      private Map<Short, CellFormatter> indexFormatterMap = new ConcurrentHashMap<>();
29      
30      /**
31       * 和暦用の日本語のロケール
32       */
33      private static final Locale LOCALE_JAPANESE = new Locale("ja", "JP", "JP");
34      
35      /**
36       * 日本語に関するロケール
37       */
38      private static final Locale[] JAPANESE_LOCALES = new Locale[]{Locale.JAPANESE, Locale.JAPAN, LOCALE_JAPANESE};
39      
40      /**
41       * 書式のパターンとフォーマッターのマップ
42       */
43      private Map<String, CellFormatter> patternFormatterMap = new ConcurrentHashMap<>();
44      
45      /**
46       * インスタンスを作成する。
47       * <p>ビルドインフォーマットなどのキャッシュ情報を予め登録する。
48       */
49      public FormatterResolver() {
50          clearFormat();
51          
52          registerDefaultFormat();
53      }
54      
55      /**
56       * キャッシュを初期化する。
57       */
58      public synchronized void clearFormat() {
59          
60          // インデックス番号指定のフォーマッタの初期化
61          indexFormatterMap.clear();
62          
63          // パターン指定の指定のフォーマッタの初期化
64          patternFormatterMap.clear();
65          
66      }
67      
68      /**
69       * キャッシュに初期値データを登録する。
70       * ・ロケールによって切り替わるフォーマットや、間違った組み込みフォーマットの場合を登録しておく。
71       */
72      public synchronized void registerDefaultFormat() {
73          
74          
75          final Locale[] availableLocales = new Locale[]{Locale.JAPANESE};
76          
77          
78          // 組み込み書式の登録
79          for(int i=0; i <= 58; i++) {
80              
81              final CellFormatter formatter = createDefaultFormatter(String.valueOf(i), availableLocales);
82              if(formatter != null) {
83                  registerFormatter((short) i, formatter);
84                  
85              }
86              
87          }
88          
89          // 特別な書式
90          final String[] names = new String[]{"F800", "F400"};
91          for(String name : names) {
92              
93              final String key = String.format("format.%s", name);
94              
95              final String defaultFormat = messageResolver.getMessage(key);
96              if(defaultFormat == null) {
97                  continue;
98              }
99              
100             final CellFormatter formatter = createDefaultFormatter(name, availableLocales);
101             if(formatter != null) {
102                 registerFormatter(defaultFormat, formatter);
103             }
104         }
105         
106     }
107     
108     /**
109      * 指定したインデックスでプロパティに定義されているフォーマットを作成する。
110      * @param name 書式の名前。({@literal format.<書式の名前>=})
111      * @param locales 検索するロケール。
112      * @return 存在しないインデックス番号の時は、nullを返す。
113      */
114     protected CellFormatter createDefaultFormatter(final String name, final Locale... locales) {
115         
116         final String key = String.format("format.%s", name);
117         
118         final String defaultFormat = messageResolver.getMessage(key);
119         if(defaultFormat == null) {
120             return null;
121         }
122         
123         CellFormatter formatter = createFormatter(defaultFormat);
124         
125         // ロケールのフォーマットの取得
126         for(Locale locale : locales) {
127             
128             final String localeFormat = messageResolver.getMessage(locale, key, null);
129             if(localeFormat == null) {
130                 continue;
131             }
132             
133             final LocaleSwitchFormatter switchFormatter;
134             if(formatter instanceof LocaleSwitchFormatter) {
135                 switchFormatter = (LocaleSwitchFormatter) formatter;
136                 
137             } else {
138                 // LocaleSwitchFormatterに入れ替える。
139                 switchFormatter = new LocaleSwitchFormatter(formatter);
140                 formatter = switchFormatter;
141             }
142             
143             // ロケールごとのフォーマットの登録
144             if(locale.equals(Locale.JAPANESE)) {
145                 switchFormatter.register(createFormatter(localeFormat), JAPANESE_LOCALES);
146                 
147             } else {
148                 switchFormatter.register(createFormatter(localeFormat), locale);
149             }
150             
151         }
152         
153         return formatter;
154         
155     }

157     /**
158      * インデックス形式の書式指定がキャッシュに登録され、解決可能かどうか。
159      * @param formatIndex 検証対象の書式のインデックス番号。
160      * @return true: 事前に登録されているものやキャッシュされているインデックス番号の場合。
161      */
162     public boolean canResolve(final short formatIndex) {
163         return indexFormatterMap.containsKey(formatIndex);
164     }
165     
166     /**
167      * パターン形式の書式指定がキャッシュに登録され、解決可能かどうか。
168      * @param formatPattern 検証対象の書式パターン。
169      * @return true:  事前に登録されているものやキャッシュされている書式パターンの場合。
170      */
171     public boolean canResolve(final String formatPattern) {
172         final String key = (formatPattern == null ? "" : formatPattern);
173         return patternFormatterMap.containsKey(key);
174     }
175     
176     /**
177      * インデックス番号を指定して、キャッシュに登録されているフォーマッタを取得する。
178      * @param formatIndex 検証対象の書式のインデックス番号。
179      * @return 登録されていないインデックスの場合は、nullを返す。
180      */
181     public CellFormatter getFormatter(final short formatIndex) {
182         
183         return indexFormatterMap.get(formatIndex);
184     }
185     
186     /**
187      * 書式のパターンを指定して、キャッシュに登録されているフォーマッタを取得する。
188      * @param formatPattern 書式
189      * @return 登録されていないインデックスの場合は、nullを返す。
190      */
191     public CellFormatter getFormatter(final String formatPattern) {
192         
193         final String key = (formatPattern == null ? "" : formatPattern);
194         return patternFormatterMap.get(key);
195     }
196     
197     /**
198      * パターンを指定して新たに書式を作成する。
199      * @param formatPattern 書式パターン。
200      * @return パースしたフォーマッタ。
201      */
202     public CellFormatter createFormatter(final String formatPattern) {
203         
204         final CellFormatter formatter = customFormatterFactory.create(formatPattern);
205         return formatter;
206         
207     }
208     
209     /**
210      * 書式インデックス番号を指定してフォーマッタをキャッシュに登録する。
211      * @param formatIndex 書式のインデックス番号。
212      * @param cellFormatter 登録対象のフォーマッタ。
213      * @return 以前に登録されたフォーマッタのインスタンス。登録されたフォーマッタがなければ、nullを返す。
214      */
215     public synchronized CellFormatter registerFormatter(final short formatIndex, final CellFormatter cellFormatter) {
216         return indexFormatterMap.put(formatIndex, cellFormatter);
217     }
218     
219     /**
220      * 書式パターンを指定してフォーマッタをキャッシュに登録する。
221      * @param formatPattern 書式パターン。
222      * @param cellFormatter 登録対象のフォーマッタ。
223      * @return 以前に登録されたフォーマッタのインスタンス。登録されたフォーマッタがなければ、nullを返す。
224      */
225     public synchronized CellFormatter registerFormatter(final String formatPattern, final CellFormatter cellFormatter) {
226         return patternFormatterMap.put(formatPattern, cellFormatter);
227     }
228     
229     /**
230      * 書式パターンを解析して、{@link CellFormatter}のインスタンスを作成するクラスの取得。
231      * @return デフォルトは標準のクラスが設定されています。
232      */
233     public CustomFormatterFactory getCustomFormatterFactory() {
234         return customFormatterFactory;
235     }
236     
237     /**
238      * 書式パターンを解析して、{@link CellFormatter}のインスタンスを作成するクラスの設定。
239      * <p>独自のカスタマイズしたものが設定可能。
240      * @param customFormatterFactory
241      */
242     public void setCustomFormatterFactory(CustomFormatterFactory customFormatterFactory) {
243         this.customFormatterFactory = customFormatterFactory;
244     }
