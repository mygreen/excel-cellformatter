1   package com.github.mygreen.cellformatter.callback;
2   
3   import java.util.Locale;
4   
5   /**
6    * 半角数字を全角数字に変換する。
7    * 
8    * @version 0.5
9    * @author T.TSUCHIE
10   *
11   */
12  public class ZenkakuNumberCallback extends JapaneseCallback<Object> {
13      
14      private static final String[][] MAP = {
15          {"0", "０"},
16          {"1", "１"},
17          {"2", "２"},
18          {"3", "３"},
19          {"4", "４"},
20          {"5", "５"},
21          {"6", "６"},
22          {"7", "７"},
23          {"8", "８"},
24          {"9", "９"},
25      };
26      
27      public static ZenkakuNumberCallback create() {
28          return new ZenkakuNumberCallback();
29      }
30      
31      @Override
32      public String call(final Object data, final String value, final Locale locale) {
33          
34          String str = value;
35          for(String[] item : MAP) {
36              str = str.replaceAll(item[0], item[1]);
37          }
38          return str;
39      }
40  }
